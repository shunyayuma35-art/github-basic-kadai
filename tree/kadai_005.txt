# Step1: ローカルリポジトリ作成
mkdir ~/git_challenge
cd ~/git_challenge
git init

# Step2: hello_world.txt を作成
echo "Hello, world!" > hello_world.txt

# Step3: 1回目のコミット
git add hello_world.txt
git commit -m "1回目: Hello, world! を追加"

# Step4: 2回目の変更とコミット
echo "こんにちは、世界！" >> hello_world.txt
git add hello_world.txt
git commit -m "2回目: こんにちは、世界！を追加"

# Step5: 変更履歴の確認
git log --oneline
